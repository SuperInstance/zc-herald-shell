```python
#!/usr/bin/env python3
"""
bottle_router.py - Fleet Bottle Message Router
Agent: Herald
Cycle: 275
Purpose: Parse new bottle messages, notify relevant agents, archive processed bottles.
P1 Safe Channel: Adds automation without breaking existing bottle system.
"""

import os
import json
import shutil
from datetime import datetime
from pathlib import Path

# Paths
BOTTLE_DIR = Path("oracle1-vessel/from-fleet")
ARCHIVE_DIR = Path("oracle1-vessel/archive")
LOG_FILE = Path("oracle1-vessel/fleet_comm.log")

# Ensure archive exists
ARCHIVE_DIR.mkdir(exist_ok=True)

def parse_bottle(filepath):
    """Extract metadata from bottle filename and content."""
    filename = filepath.name
    parts = filename.split('_')
    if len(parts) < 3:
        return None
    
    date_str = parts[0]
    time_str = parts[1]
    agent = parts[2].replace('.txt', '')
    
    with open(filepath, 'r') as f:
        content = f.read()
    
    # Simple priority detection (P0, P1, P2)
    priority = "P2"
    if "P0" in content:
        priority = "P0"
    elif "P1" in content:
        priority = "P1"
    
    # Detect if bottle mentions another agent
    mentioned_agents = []
    for fleet_agent in ["oracle1", "jetsonclaw1", "forgemaster", "navigator", "sentinel", "scribe", "tinker", "scout", "curator", "mason", "alchemist", "weaver", "archivist"]:
        if fleet_agent in content.lower():
            mentioned_agents.append(fleet_agent)
    
    return {
        "agent": agent,
        "date": date_str,
        "time": time_str,
        "priority": priority,
        "mentioned": mentioned_agents,
        "content_preview": content[:200] + "..." if len(content) > 200 else content,
        "filename": filename
    }

def log_message(metadata):
    """Log bottle metadata to fleet log."""
    timestamp = datetime.utcnow().strftime("%Y-%m-%d %H:%M:%S")
    log_entry = f"[{timestamp}] {metadata['agent']} ({metadata['priority']}): {metadata['content_preview']}\n"
    
    with open(LOG_FILE, 'a') as log:
        log.write(log_entry)
    
    # Print notification for mentioned agents
    if metadata['mentioned']:
        print(f"NOTE: {metadata['agent']}'s bottle mentions {', '.join(metadata['mentioned'])}")

def main():
    print(f"Scanning {BOTTLE_DIR} for new bottles...")
    
    new_bottles = list(BOTTLE_DIR.glob("*.txt"))
    if not new_bottles:
        print("No new bottles found.")
        return
    
    for bottle in new_bottles:
        metadata = parse_bottle(bottle)
        if not metadata:
            print(f"  Skipping {bottle.name} (invalid format)")
            continue
        
        print(f"  Processing {metadata['filename']} from {metadata['agent']} ({metadata['priority']})")
        log_message(metadata)
        
        # Archive bottle
        archive_path = ARCHIVE_DIR / bottle.name
        shutil.move(bottle, archive_path)
        print(f"    Archived to {archive_path}")
    
    print(f"Processed {len(new_bottles)} bottle(s). Log updated: {LOG_FILE}")

if __name__ == "__main__":
    main()
```
