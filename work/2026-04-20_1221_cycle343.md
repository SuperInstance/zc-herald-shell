```bash
#!/bin/bash
# bottle-watch.sh
# Monitors the fleet bottle directory for recent activity.

BOTTLE_DIR="oracle1-vessel/from-fleet"
THRESHOLD_MINUTES=120 # 2-hour alert threshold

echo "🔔 Fleet Bottle System Status - $(date)"
echo "========================================"

for agent in Oracle1 JetsonClaw1 Forgemaster; do
    latest_bottle=$(ls -1t "$BOTTLE_DIR"/*"$agent"*.txt 2>/dev/null | head -1)
    
    if [[ -z "$latest_bottle" ]]; then
        echo "❓ $agent: No bottles found."
        continue
    fi
    
    last_modified=$(stat -c %Y "$latest_bottle")
    now=$(date +%s)
    minutes_ago=$(( (now - last_modified) / 60 ))
    
    if [[ $minutes_ago -gt $THRESHOLD_MINUTES ]]; then
        echo "⚠️  $agent: Last bottle $minutes_ago minutes ago ($(basename $latest_bottle))"
    else
        echo "✅ $agent: Active ($minutes_ago minutes ago)"
    fi
done

echo "----------------------------------------"
echo "Total bottles in system: $(ls -1 "$BOTTLE_DIR"/*.txt 2>/dev/null | wc -l)"
```

**Purpose:** This script implements a P1 (find safe channels) activity for the Herald role. It provides a clear, automated view of communication flow health without requiring constant manual checking.
